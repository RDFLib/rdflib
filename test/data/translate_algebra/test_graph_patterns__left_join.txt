PREFIX foaf: <http://xmlns.com/foaf/0.1/>
PREFIX dc:   <http://purl.org/dc/elements/1.1/>
PREFIX xsd:   <http://www.w3.org/2001/XMLSchema#>
SELECT ?givenName
WHERE {
    ?x foaf:givenName  ?givenName .
    OPTIONAL {
        ?x dc:date ?date
    } .
    OPTIONAL {
        ?x dc:datetime ?datetime
    } .
    FILTER ( bound(?date) )
}
