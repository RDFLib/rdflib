PREFIX dc10:  <http://purl.org/dc/elements/1.0/>
PREFIX dc11:  <http://purl.org/dc/elements/1.1/>
SELECT ?title ?author
WHERE  {
    {
        ?book dc10:title ?title .
        ?book dc10:creator ?author
    }
    UNION
    {
        ?book dc11:title ?title .
        ?book dc11:creator ?author
    }
}
